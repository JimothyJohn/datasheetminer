Quick Start

# Install dependencies
uv sync

# Set your Gemini API key
export GEMINI_API_KEY="your-api-key"

# Analyze a document
uv run datasheetminer/scraper.py \
  --type motor \
  --url https://example.com/datasheet.pdf \
  --pages 1-5 \
  --output specs.json
